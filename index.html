// ── shared tiny components ──────────────────────────────────────────────

function Avatar({ label, size=36, colorIdx=0 }) {

  return <div style={{ width:size, height:size, borderRadius:"50%", background:getColor(colorIdx), display:"flex", alignItems:"center", justifyContent:"center", color:"#fff", fontWeight:500, fontSize:size*0.35, flexShrink:0 }}>{label}</div>;

}


function Badge({ children, bg, color }) {

  return <span style={{ background:bg, color, borderRadius:20, padding:"3px 10px", fontSize:11, fontWeight:600 }}>{children}</span>;

}


function SectionTitle({ children }) {

  return <div style={{ fontWeight:600, fontSize:13, color:"#334155", marginBottom:10 }}>{children}</div>;

}


function Card({ children, style={} }) {

  return <div style={{ background:"#fff", border:"0.5px solid #e2e8f0", borderRadius:12, padding:"14px", marginBottom:10, ...style }}>{children}</div>;

}


function Toast({ toasts }) {

  return (

    <div style={{ position:"fixed", bottom:80, right:16, zIndex:999, display:"flex", flexDirection:"column", gap:8 }}>

      {toasts.map(t => (

        <div key={t.id} style={{ background: t.type==="success"?"#0F6E56":t.type==="warn"?"#854F0B":"#185FA5", color:"#fff", padding:"10px 14px", borderRadius:10, fontSize:12, fontWeight:500, maxWidth:260, display:"flex", gap:8, alignItems:"center" }}>

          <span>{t.type==="success"?"✓":t.type==="warn"?"⏰":"ℹ"}</span>{t.msg}

        </div>

      ))}

    </div>

  );

}


function Modal({ show, onClose, title, subtitle, children }) {

  if (!show) return null;

  return (

    <div style={{ position:"fixed", inset:0, background:"rgba(0,0,0,0.5)", display:"flex", alignItems:"flex-end", justifyContent:"center", zIndex:200 }}>

      <div style={{ background:"#fff", borderRadius:"20px 20px 0 0", padding:"24px 20px 32px", width:"100%", maxWidth:480, maxHeight:"90vh", overflowY:"auto" }}>

        <div style={{ width:36, height:4, background:"#e2e8f0", borderRadius:4, margin:"0 auto 20px" }} />

        <div style={{ fontWeight:700, fontSize:16, marginBottom:4 }}>{title}</div>

        {subtitle && <div style={{ fontSize:12, color:"#64748b", marginBottom:18 }}>{subtitle}</div>}

        {children}

      </div>

    </div>

  );

function CaregiversScreen({ caregivers, setCaregivers, addToast, addNotif }) {

  const [showAdd, setShowAdd] = useState(false);

  const [editNudge, setEditNudge] = useState(null);

  const [removeConfirm, setRemoveConfirm] = useState(null);

  const [form, setForm] = useState({ name:"", email:"", relation:"", nudge:10 });

  const [formErr, setFormErr] = useState({});


  const validate = () => {

    const err={};

    if (!form.name.trim()) err.name="Required";

    if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(form.email)) err.email="Valid email required";

    if (!form.relation.trim()) err.relation="Required";

    setFormErr(err); return !Object.keys(err).length;

  };


  const add = () => {

    if (!validate()) return;

    const av = form.name.trim().split(" ").map(w=>w[0]).join("").slice(0,2).toUpperCase();

    const cg = { id:"c"+Date.now(), ...form, name:form.name.trim(), email:form.email.trim(), relation:form.relation.trim(), active:true, avatar:av };

    setCaregivers(p=>[...p,cg]);

    setForm({ name:"", email:"", relation:"", nudge:10 }); setFormErr({});

    setShowAdd(false);

    addToast(`${cg.name} added`,"success");

    addNotif(`Caregiver added: ${cg.name} (${cg.email})`);

  };


  const remove = id => {

    const cg = caregivers.find(c=>c.id===id);

    setCaregivers(p=>p.filter(c=>c.id!==id));

    setRemoveConfirm(null);

    addToast(`${cg.name} removed`,"warn");

    addNotif(`Caregiver removed: ${cg.name}`);

  };


  const updateNudge = (id,val) => {

    setCaregivers(p=>p.map(c=>c.id===id?{...c,nudge:val}:c));

    setEditNudge(null);

    addToast("Nudge timing updated","success");

  };


  const toggle = id => {

    const cg = caregivers.find(c=>c.id===id);
